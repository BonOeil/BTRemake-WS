FROM mcr.microsoft.com/dotnet/sdk:7.0 AS build
WORKDIR /source

# Copier les fichiers projet et restaurer les dépendances
COPY *.csproj .
RUN dotnet restore

# Copier le reste du code et compiler
COPY . .
RUN dotnet publish -c Release -o /app

FROM mcr.microsoft.com/dotnet/aspnet:7.0
WORKDIR /app
COPY --from=build /app .

# Variables d'environnement pour la configuration
ENV ServerSettings__Port=5000
ENV ServerSettings__MaxPlayers=50
ENV ServerSettings__AllowedOrigins__0=http://localhost:8080 
ENV ServerSettings__AllowedOrigins__1=https://votredomaine.com

EXPOSE 5000

ENTRYPOINT ["dotnet", "GameServer.dll"]
